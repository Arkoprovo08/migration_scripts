BEGIN
   FOR rec IN (
      SELECT DISTINCT REF_ID 
      FROM FRAMEWORK01.FORM_LD_CALCULATIONS 
      WHERE STATUS = '1' AND SEQ = 1 AND IS_MIGRATED_02 = 0
   )
   LOOP
      BEGIN
         migrate_lqd_damages_02(rec.REF_ID);
      EXCEPTION
         WHEN OTHERS THEN
            log_error(SQLCODE, SQLERRM, 'LD MIGRATION LOOP', 'FORM_LD_CALCULATIONS', rec.REF_ID, DBMS_UTILITY.FORMAT_ERROR_BACKTRACE);
      END;
   END LOOP;
END;
/
