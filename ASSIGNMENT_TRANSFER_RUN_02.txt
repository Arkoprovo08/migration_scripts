BEGIN
   FOR rec IN (
      SELECT DISTINCT REF_ID 
      FROM FRAMEWORK01.FORM_ASSIGNMENT_INTEREST 
      WHERE STATUS = '1' AND IS_MIGRATED_02 = 0
   )
   LOOP
      BEGIN
         migrate_assignment_interest_02(rec.REF_ID);
      EXCEPTION
         WHEN OTHERS THEN
            log_error(SQLCODE, SQLERRM, 'ASSIGNMENT TRANSFER MIGRATION LOOP', 'FORM_ASSIGNMENT_INTEREST', rec.REF_ID, DBMS_UTILITY.FORMAT_ERROR_BACKTRACE);
      END;
   END LOOP;
END;
/
