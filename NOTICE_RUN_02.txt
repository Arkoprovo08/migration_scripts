BEGIN
   FOR rec IN (
      SELECT DISTINCT REFID 
      FROM FRAMEWORK01.FORM_NOTICE_NEXT_PHASE_RELIN 
      WHERE IS_MIGRATED_02 = 0
   )
   LOOP
      BEGIN
         migrate_notice_next_phase_02(rec.REFID);
      EXCEPTION
         WHEN OTHERS THEN
            log_error(SQLCODE, SQLERRM, 'NOTICE NEXT PHASE MIGRATION LOOP', 'FORM_NOTICE_NEXT_PHASE_RELIN', rec.REFID, DBMS_UTILITY.FORMAT_ERROR_BACKTRACE);
      END;
   END LOOP;
END;
